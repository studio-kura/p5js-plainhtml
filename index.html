//オセロ 一回目。今回はオセロ のマス目を線で引く。コマをマウスプレスで置く。ところまで行ってみたいと思います。 let sketch = function(p) { let col; // 横方向に何マス目に円を描くか let row; // 縦方向に何マス目に円を描くか p.setup = function() { var canvas = p.createCanvas(400, 400); canvas.parent('sketch-container'); } p.draw = function() { p.background(0,140,0); //for文の中にfor文を記述し線を縦横に８本ずつ描いてください。 for(let i = 0; i < 8; i++){ for(let j = 0; j < 8; j++){ p.line(j * 50, 0, j * 50, p.height); p.line(0, i * 50, p.width, i * 50); //升目状に丸が表示される様にしましょう if(j == col && i == row){ p.fill(255); p.ellipse(j * 50 + 25, i* 50 + 25, 40,40); } } } } p.mousePressed = function() { // クリックした時のマウス座標をマスの長さで割って切り捨てると何マス目かがわかる col = p.floor(p.mouseX / 50); row = p.floor(p.mouseY / 50); } }; let myp5 = new p5(sketch);
